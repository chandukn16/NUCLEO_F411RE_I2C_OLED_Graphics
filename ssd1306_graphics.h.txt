/*
 * ssd1306_graphics.h
 *
 * Simple graphics primitives for SSD1306 framebuffer driver (uses SSD1306_DrawPixel)
 *
 * Usage:
 *  - Add this file and ssd1306_graphics.c to your project.
 *  - Make sure ssd1306.{h,c} from the previous message is included and initialized.
 *  - Call the primitives, then call SSD1306_UpdateScreen() to show the framebuffer.
 *
 * Functions:
 *  - SSD1306_DrawLine        : Bresenham line
 *  - SSD1306_DrawRect        : Rectangle outline
 *  - SSD1306_FillRect        : Filled rectangle
 *  - SSD1306_DrawCircle      : Circle outline (midpoint)
 *  - SSD1306_FillCircle      : Filled circle
 *  - SSD1306_DrawBitmapXY    : Draw 1-bit-per-pixel bitmap stored as rows of bytes (MSB first)
 */

#ifndef INC_SSD1306_GRAPHICS_H_
#define INC_SSD1306_GRAPHICS_H_

#include <stdint.h>
#include "ssd1306.h"

void SSD1306_DrawLine(int16_t x0, int16_t y0, int16_t x1, int16_t y1, uint8_t color);
void SSD1306_DrawRect(int16_t x, int16_t y, int16_t w, int16_t h, uint8_t color);
void SSD1306_FillRect(int16_t x, int16_t y, int16_t w, int16_t h, uint8_t color);
void SSD1306_DrawCircle(int16_t x0, int16_t y0, int16_t r, uint8_t color);
void SSD1306_FillCircle(int16_t x0, int16_t y0, int16_t r, uint8_t color);

/*
 * Draw a monochrome bitmap.
 * bitmap: pointer to bytes; each row is packed MSB-first. Bytes-per-row = (bw + 7) / 8.
 * bw: bitmap width in pixels
 * bh: bitmap height in pixels
 * (x,y) is the top-left position on the display
 */
void SSD1306_DrawBitmapXY(int16_t x, int16_t y, const uint8_t *bitmap, int16_t bw, int16_t bh, uint8_t color);

#endif /* INC_SSD1306_GRAPHICS_H_ */
